1) select NameOfAirport from Airport

2) select FullName from Pilot,Airport ap
   where Airport=ap.OID and ap.NameOfAirport='À'

3) select NameOfPlane from Plane,Airport ap
   where Airport=ap.OID and ap.NameOfAirport='À'

4) select NameOfPlane from Plane p,Pilot pl,PlanePlanes_PilotPilots
   where pl.FullName='P' and pl.OID=Pilots and p.OID=Planes

5) select FullName from Pilot pl,Plane p 
   where p.NameOfPlane='S' and p.Airport=pl.Airport

6) select NameOfAirport from (select NameOfAirport, Count(*) as count from  (select NameOfAirport,FullName from Airport,Pilot
   where Pilot.Airport=Airport.OID) x group by NameOfAirport) y 
   where y.count>10
